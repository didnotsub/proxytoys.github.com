011    package com.thoughtworks.proxy.kit;
012    
013    import java.io.Serializable;
014    
015    /**
016     * Simple implementation for an {@link ObjectReference}.
017     *
018     * @author Aslak Helles&oslash;y
019     * @since 0.2
020     */
021    public class SimpleReference<T> implements ObjectReference<T>, Serializable {
022        private static final long serialVersionUID = 1L;
023        private T instance;
024    
025        /**
026         * Construct a SimpleReference.
027         *
028         * @param reference The referenced object.
029         * @since 0.2
030         */
031        public SimpleReference(final T reference) {
032            set(reference);
033        }
034    
035        public T get() {
036            return instance;
037        }
038    
039        public void set(final T item) {
040            this.instance = item;
041        }
042    }
