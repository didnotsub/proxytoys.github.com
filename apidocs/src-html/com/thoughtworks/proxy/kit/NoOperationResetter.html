011    package com.thoughtworks.proxy.kit;
012    
013    import java.io.Serializable;
014    
015    
016    /**
017     * A Resetter that has no operation. This Resetter implementation will just return the object to the pool without
018     * further interaction.
019     *
020     * @author J&ouml;rg Schaible
021     * @since 0.2
022     */
023    public class NoOperationResetter<T> implements Resetter<T>, Serializable {
024    
025        private static final long serialVersionUID = 1L;
026    
027        /**
028         * {@inheritDoc} Returns just <code>true</code>.
029         */
030        public boolean reset(T object) {
031            return true;
032        }
033    
034    }
